#!/bin/bash

WORK_DIR="/var/local/PackageMonitor"
INDEX_DIR="$WORK_DIR/index_data"
LOG_SOURCE="/var/log/dpkg.log"

echo "--- Initializare PackageMonitor ---"

if ! mkdir -p "$INDEX_DIR" 2>/dev/null; then
    echo "Eroare: Nu am permisiuni de scriere in /var/local. Ruleaza scriptul cu sudo!"
    exit 1
fi

[ ! -f "$WORK_DIR/total_size.dat" ] && echo "0" > "$WORK_DIR/total_size.dat"
[ ! -f "$WORK_DIR/undo_cache.log" ] && touch "$WORK_DIR/undo_cache.log"
[ ! -f "$WORK_DIR/monitor_status.dat" ] && echo "IDLE" > "$WORK_DIR/monitor_status.dat"

if [ ! -f "$LOG_SOURCE" ]; then
    echo "Eroare: Fisierul $LOG_SOURCE nu a fost gasit!"
    exit 1
fi

echo "Introduceti perioada pentru analiza (Format: AAAA-LL-ZZ):"
read -p "Data de inceput: " START_DATE
read -p "Data de sfarsit: " END_DATE

if [[ ! $START_DATE =~ ^[0-9]{4}-[0-9]{2}-[0-9]{2}$ ]]; then
    echo "Eroare: Formatul datei este invalid!"
    exit 1
fi

echo "Se proceseaza log-ul intre $START_DATE si $END_DATE..."

grep -E "install |remove " "$LOG_SOURCE" | while read -r line; do

    DATA=$(echo "$line" | awk '{print $1}')
    ORA=$(echo "$line" | awk '{print $2}')
    ACTIUNE=$(echo "$line" | awk '{print $3}')
    PACHET=$(echo "$line" | awk '{print $4}')
    V_VECHE=$(echo "$line" | awk '{print $5}')
    V_NOUA=$(echo "$line" | awk '{print $6}')

    if [[ "$DATA" < "$START_DATE" ]] || [[ "$DATA" > "$END_DATE" ]]; then
        continue
    fi

    PKG_DIR="$WORK_DIR/$PACHET"
    mkdir -p "$PKG_DIR"

    echo "$line" >> "$PKG_DIR/istoric.log"
    echo "ULTIMA_ACTIUNE=$ACTIUNE | DATA=$DATA | ORA=$ORA | VERSIUNE=$V_NOUA" > "$PKG_DIR/stare_curenta.dat"
    echo "$PACHET" >> "$INDEX_DIR/${DATA}_${ACTIUNE}s.txt"

    if [ "$ACTIUNE" == "install" ]; then

        SIZE=$(dpkg-query -W -f='${Installed-Size}' "$PACHET" 2>/dev/null || echo "0")
        if [ "$SIZE" -gt 0 ]; then
            TOTAL_VECHI=$(cat "$WORK_DIR/total_size.dat")
            TOTAL_NOU=$((TOTAL_VECHI + SIZE))
            echo "$TOTAL_NOU" > "$WORK_DIR/total_size.dat"
        fi
    fi

    if [ "$ACTIUNE" == "remove" ]; then

        echo "$PACHET ($DATA $ORA)" > "$WORK_DIR/undo_cache.tmp"
        cat "$WORK_DIR/undo_cache.log" >> "$WORK_DIR/undo_cache.tmp" 2>/dev/null
        head -n 10 "$WORK_DIR/undo_cache.tmp" > "$WORK_DIR/undo_cache.log"
        rm "$WORK_DIR/undo_cache.tmp"
    fi

done

echo "Monitorizarea s-a incheiat cu succes. Datele sunt in $WORK_DIR."
echo "STATUS: FINISHED" > "$WORK_DIR/monitor_status.dat"
